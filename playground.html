FizzBuzz (Python Style)

# Python: for i in range(1, n+1):
#     print("FizzBuzz" if i % 15 == 0 else "Fizz" if i % 3 == 0 else "Buzz" if i % 5 == 0 else i)
function fizzBuzz() {
    let n = Number(document.getElementById('fizzInput').value);
    let result = '';
    for (let i = 1; i <= n; i++) {
        result += (i % 15 === 0 ? 'FizzBuzz' : i % 3 === 0 ? 'Fizz' : i % 5 === 0 ? 'Buzz' : i) + '
';
    }
    document.getElementById('fizzResult').innerHTML = result;
}

Prime Checker (Python Style)

# Python: def is_prime(n):
#     if n < 2: return False
#     for i in range(2, int(n ** 0.5) + 1):
#         if n % i == 0: return False
#     return True
function checkPrime() {
    let n = Number(document.getElementById('primeInput').value);
    let isPrime = n >= 2;
    for (let i = 2; i <= Math.sqrt(n) && isPrime; i++) {
        if (n % i === 0) isPrime = false;
    }
    document.getElementById('primeResult').innerText = isPrime ? `${n} is prime!` : `${n} is not prime.`;
}

Memory Game (C Style)

// C: int arr[4]; // Manual memory allocation
// Simulate memory with an array and random values
function startMemoryGame() {
    let sequence = Array(4).fill().map(() => Math.floor(Math.random() * 4) + 1);
    let display = document.getElementById('memoryDisplay');
    display.innerHTML = sequence.join(' ') + '
Memorize this!';
    setTimeout(() => {
        display.innerText = 'Guess the sequence!';
        document.getElementById('memoryInput').style.display = 'block';
        document.getElementById('memorySubmit').style.display = 'block';
        window.currentSequence = sequence;
    }, 2000);
}
function checkMemory() {
    let guess = document.getElementById('memoryInput').value.split(' ').map(Number);
    let result = guess.join('') === window.currentSequence.join('') ? 'Correct!' : 'Wrong!';
    document.getElementById('memoryResult').innerText = result;
}
